{{- $ctx := .ctx | default . -}} {{- /* 最终的页面组装约定： - primary_nav: 站点顶级导航；值必须是 fragment page 路径 - utilities: Me / Language / Theme；值必须是 fragment page 路径 - breadcrumb_root: 面包屑首字段 / 根域切换；值必须是 breadcrumb model fragment 路径 - breadcrumb: 当前页面路径尾部；只接受 true/false - meta: 当前页面元信息（路径、taxonomy、日期）；只接受 true/false - footer: 页脚 fragment；值必须是 fragment page 路径 补充规则： - main 是框架内建 slot，不通过 front matter 指向 fragment - 不再支持 show_breadcrumb / show_meta / breadcrumb_variant - 不再使用旧的 context 命名；breadcrumb_root 即原“根域切换”语义 */ -}} {{- $slotConfig := dict -}} {{- with index $ctx.Params "slots" -}} {{- if not (reflect.IsMap .) -}} {{- errorf "Build failed: page %q must define slots as a map/object." ($ctx.File.Path | default (printf "%s %s" $ctx.Kind $ctx.RelPermalink)) -}} {{- end -}} {{- $slotConfig = . -}} {{- end -}} {{- $supportedSlots := slice "primary_nav" "breadcrumb_root" "utilities" "breadcrumb" "meta" "footer" -}} {{- $supportedLookup := dict -}} {{- range $slot := $supportedSlots -}} {{- $supportedLookup = merge $supportedLookup (dict $slot true) -}} {{- end -}} {{- range $slotName, $_ := $slotConfig -}} {{- $key := printf "%v" $slotName -}} {{- if not (isset $supportedLookup $key) -}} {{- errorf "Build failed: page %q does not support slots.%s." ($ctx.File.Path | default (printf "%s %s" $ctx.Kind $ctx.RelPermalink)) $key -}} {{- end -}} {{- end -}} {{- $sources := dict -}} {{- range $slot := $supportedSlots -}} {{- $sources = merge $sources (dict $slot false) -}} {{- if not (isset $slotConfig $slot) -}}{{- continue -}}{{- end -}} {{- $raw := index $slotConfig $slot -}} {{- if eq $raw false -}}{{- continue -}}{{- end -}} {{- if eq $slot "breadcrumb" -}} {{- if eq $raw true -}} {{- $sources = merge $sources (dict $slot true) -}} {{- continue -}} {{- end -}} {{- errorf "Build failed: page %q only supports true/false for slots.breadcrumb." ($ctx.File.Path | default (printf "%s %s" $ctx.Kind $ctx.RelPermalink)) -}} {{- end -}} {{- if eq $slot "meta" -}} {{- if eq $raw true -}} {{- $sources = merge $sources (dict $slot true) -}} {{- continue -}} {{- end -}} {{- errorf "Build failed: page %q only supports true/false for slots.meta." ($ctx.File.Path | default (printf "%s %s" $ctx.Kind $ctx.RelPermalink)) -}} {{- end -}} {{- $src := trim (printf "%v" $raw) " \r\n\t" -}} {{- if eq $src "" -}}{{- continue -}}{{- end -}} {{- $source := $ctx.Site.GetPage $src -}} {{- if not $source -}} {{- errorf "Build failed: page %q cannot resolve slots.%s=%q to a page." ($ctx.File.Path | default (printf "%s %s" $ctx.Kind $ctx.RelPermalink)) $slot $src -}} {{- end -}} {{- $sources = merge $sources (dict $slot $source) -}} {{- partial "slots/slot-capabilities-register.html" (dict "ctx" $ctx "slot" $slot "source_ref" $src) -}} {{- end -}} {{- $ctx.Scratch.Set "slot_sources" $sources -}} {{- return $sources -}}
