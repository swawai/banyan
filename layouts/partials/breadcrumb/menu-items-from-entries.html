{{- $entries := .entries | default slice -}} {{- $selectedHref := printf "%v" (.selected_href | default "") -}} {{- $textKey := printf "%v" (.text_key | default "title") -}} {{- $hrefKey := printf "%v" (.href_key | default "href") -}} {{- $items := slice -}} {{- range $entry := $entries -}} {{- $text := "" -}} {{- with index $entry $textKey -}} {{- $text = printf "%v" . -}} {{- end -}} {{- $href := "" -}} {{- with index $entry $hrefKey -}} {{- $href = printf "%v" . -}} {{- end -}} {{- if or (eq $text "") (eq $href "") -}} {{- continue -}} {{- end -}} {{- $item := dict "text" $text "href" $href "current" (eq $href $selectedHref) -}} {{- with index $entry "kind" -}} {{- $kind := lower (trim (printf "%v" .) " ") -}} {{- if ne $kind "" -}}{{- $item = merge $item (dict "kind" $kind) -}}{{- end -}} {{- end -}} {{- $items = $items | append $item -}} {{- end -}} {{- return $items -}}
