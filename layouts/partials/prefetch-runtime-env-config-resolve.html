{{- $prefetchLabels := .prefetch_labels | default (slice "menu" "nav" "sort_siblings" "descendants" "post") -}} {{- $envKey := upper (printf "%v" (.env_key | default "")) -}} {{- $envConfigMap := .env_config_map | default dict -}} {{- $stack := .stack | default (slice) -}} {{- if in $stack $envKey -}} {{- errorf "Build failed: params.prefetch_runtime.env.%s reference chain contains a cycle: %s." $envKey (delimit (append $stack $envKey) " -> ") -}} {{- end -}} {{- $rawEnvConfig := index $envConfigMap $envKey | default dict -}} {{- $refTargetRaw := "" -}} {{- if isset $rawEnvConfig "ref" -}} {{- $refTargetRaw = printf "%v" (index $rawEnvConfig "ref") -}} {{- end -}} {{- $refTargetRaw = replaceRE "^\\s+|\\s+$" "" $refTargetRaw -}} {{- $resolvedEnvConfig := dict -}} {{- if and (ne $refTargetRaw "") (ne $refTargetRaw "") -}} {{- $refTarget := upper $refTargetRaw -}} {{- if eq (len (findRE "^[TF]{2}$" $refTarget)) 0 -}} {{- errorf "Build failed: params.prefetch_runtime.env.%s reference targets only support two-flag T/F environment keys (for example TT, FT, FF); got %q." $envKey $refTargetRaw -}} {{- end -}} {{- if not (isset $envConfigMap $refTarget) -}} {{- errorf "Build failed: params.prefetch_runtime.env.%s references undefined environment %q." $envKey $refTarget -}} {{- end -}} {{- $resolvedEnvConfig = partial "prefetch-runtime-env-config-resolve.html" (dict "env_key" $refTarget "env_config_map" $envConfigMap "prefetch_labels" $prefetchLabels "stack" ($stack | append $envKey) ) -}} {{- end -}} {{- range $label := $prefetchLabels -}} {{- if isset $rawEnvConfig $label -}} {{- $resolvedEnvConfig = merge $resolvedEnvConfig (dict $label (index $rawEnvConfig $label)) -}} {{- end -}} {{- end -}} {{- return $resolvedEnvConfig -}}
