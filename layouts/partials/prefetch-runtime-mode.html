{{- $rawValue := lower (printf "%v" . | replaceRE "^\\s+|\\s+$" "") -}} {{- $raw := $rawValue -}} {{- $globalGate := false -}} {{- if gt (len (findRE "_g$" $raw)) 0 -}} {{- $globalGate = true -}} {{- $raw = replaceRE "_g$" "" $raw -}} {{- end -}} {{- $result := dict "kind" "invalid" "transport" "invalid" "global_gate" false "value" $rawValue -}} {{- if or (eq $rawValue "") (eq $rawValue "off") (eq $rawValue "") -}} {{- $result = dict "kind" "" "transport" "" "global_gate" false "value" $rawValue -}} {{- else -}} {{- $match := findRE "^(link|sw)_([smx])f$" $raw -}} {{- if gt (len $match) 0 -}} {{- $parts := split $raw "_" -}} {{- if eq (len $parts) 2 -}} {{- $transport := index $parts 0 -}} {{- $modeCode := index $parts 1 -}} {{- $eagernessCode := substr $modeCode 0 1 -}} {{- $eagerness := "" -}} {{- if eq $eagernessCode "s" -}} {{- $eagerness = "conservative" -}} {{- else if eq $eagernessCode "m" -}} {{- $eagerness = "moderate" -}} {{- else if eq $eagernessCode "x" -}} {{- $eagerness = "eager" -}} {{- end -}} {{- if and $transport $eagerness -}} {{- $result = dict "kind" "prefetch" "eagerness" $eagerness "transport" $transport "global_gate" $globalGate "value" $rawValue -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- return $result -}}
